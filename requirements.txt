
import os
import json
from cryptography.fernet import Fernet

# =========================================================
# CONFIGURAÇÃO DE SEGURANÇA E CRIPTOGRAFIA
# =========================================================

# Caminho do arquivo que armazena a chave mestra
FERNET_KEY_FILE = "fernet_master.key"

# Gera a chave mestra se não existir
if not os.path.exists(FERNET_KEY_FILE):
    master_key = Fernet.generate_key()
    with open(FERNET_KEY_FILE, "wb") as f:
        f.write(master_key)
    print("Chave mestra gerada! Guarde este valor:", master_key.decode())
else:
    with open(FERNET_KEY_FILE, "rb") as f:
        master_key = f.read()

fernet = Fernet(master_key)

# =========================================================
# SUAS CHAVES SENSÍVEIS
# =========================================================
# Substitua pelos valores reais uma vez, depois eles serão criptografados
secrets_plain = {
    "INSTAGRAM_TOKEN": "SUA_INSTAGRAM_TOKEN_AQUI",
    "GEMINI_API_KEY": "SUA_GEMINI_KEY_AQUI",
    "GPT_API_KEY": "SUA_GPT_KEY_AQUI",
    "INSTAGRAM_ID": "SEU_INSTAGRAM_ID_AQUI"
}

# Arquivo que armazena chaves criptografadas
ENCRYPTED_FILE = "secrets_encrypted.json"

# Criptografa e salva automaticamente se não existir
if not os.path.exists(ENCRYPTED_FILE):
    encrypted = {k: fernet.encrypt(v.encode()).decode() for k, v in secrets_plain.items()}
    with open(ENCRYPTED_FILE, "w") as f:
        json.dump(encrypted, f, indent=4)
    print("Chaves criptografadas e salvas em", ENCRYPTED_FILE)

# =========================================================
# LEITURA AUTOMÁTICA DAS CHAVES CRIPTOGRAFADAS
# =========================================================
with open(ENCRYPTED_FILE, "r") as f:
    encrypted_data = json.load(f)

# Decryptar automaticamente
secrets = {k: fernet.decrypt(v.encode()).decode() for k, v in encrypted_data.items()}

# Agora 'secrets' tem todas as chaves decodificadas, prontas para uso
INSTAGRAM_TOKEN = secrets["INSTAGRAM_TOKEN"]
GEMINI_API_KEY = secrets["GEMINI_API_KEY"]
GPT_API_KEY = secrets["GPT_API_KEY"]
INSTAGRAM_ID = secrets["INSTAGRAM_ID"]
